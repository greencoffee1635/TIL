## HTML 태그 연습

```html
<!-- index.html -->
<!DOCTYPE html>
<html>
  <head>
    <title>Hello, WEB</title>
    <meta charset="utf-8" />
  </head>
  <body>
    <!-- 페이지 이동 -->
    <h1><a href="index.html">WEB</a></h1>
    <ul>
      <li><a href="1.html">html</a></li>
      <li><a href="2.html">css</a></li>
      <li><a href="3.html">javascript</a></li>
    </ul>

    Hello, WEB!<br /><br />
    <!-- 강조(밑줄) -->
    WWW is <strong>World <u>Wide</u> Web!</strong> <br /><br />

    <!-- 랜덤 이미지 추가 -->
    <image
      src="https://source.unsplash.com/1600x900/?water,nature"
      alt="image"
      width="300px"
    /><br /><br />
    <!-- 유튜브 영상 추가 -->
    <iframe
      width="560"
      height="315"
      src="https://www.youtube.com/embed/xpWYJIC-Z9c"
      title="YouTube video player"
      frameborder="0"
      allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
      allowfullscreen
    ></iframe>
  </body>
</html>
```
